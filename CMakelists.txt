cmake_minimum_required(VERSION 3.11)
project(pac_man CXX)

# Require modern C++
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Get sources
file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS
    src/*.cpp
)

#### Setup Directories ####
#Main output directory
SET(OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin/")
# Ouput all DLLs from all libs into main build folder
SET(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${OUTPUT_DIRECTORY})

#### Add External Dependencies ####
add_subdirectory("lib/SFML")
set(SFML_INCS "lib/SFML/include")
link_directories("${CMAKE_BINARY_DIR}/lib/SFML/lib")
add_library(${PROJECT_NAME} STATIC engine/game_system.cpp)
target_include_directories(${PROJECT_NAME} INTERFACE engine)
target_link_libraries(${PROJECT_NAME} sfml-graphics)

#### Practical 5 ####
add_executable(${PROJECT_NAME} ${SOURCES})
target_include_directories(${PROJECT_NAME} PRIVATE )
target_include_directories(${PROJECT_NAME} PRIVATE ${SFML_INCS})
target_include_directories(${PROJECT_NAME} PRIVATE ${CMAKE_SOURCE_DIR}/private)


# Resources File
file(COPY resources DESTINATION ${CMAKE_BINARY_DIR})
